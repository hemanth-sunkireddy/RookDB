1use std::fs::OpenOptions;
2use std::io;
3
4mod disk;
5mod raw_page;
6use crate::disk::{create_page, read_page, write_page};
7use crate::raw_page::{Page, page_add_data};
8
9pub const CATALOG_PATH: &str = "database/global/catalog.dat"; // Catalog file path
10
11fn main() -> io::Result<()> {
12    println!("Welcome to Storage Manager");
13
14    // Create File Pointer
15    let mut file_pointer = OpenOptions::new()
16        .read(true)
17        .write(true)
18        .open(CATALOG_PATH)?;
19
20    /*
21    Create a Page in file
22    */
23    create_page(&mut file_pointer)?;
24    println!("Page created successfully.");
25
26    // Create a Page in Memory
27    let mut page: Page = Page::new();
28
29    let content = b"Hello, Storage Manager!";
30    page.data[..content.len()].copy_from_slice(content);
31
32    let page_num: u32 = 0;
33
34    // Write page to file
35    write_page(&mut file_pointer, &mut page, page_num)?;
36    println!("Updated Page with content Successfully.");
37
38    /*
39    Reading a Page from Disk file to Memory Page
40    File: Catalog
41    Page: page
42    PageNum: 0
43    */
44    read_page(&mut file_pointer, &mut page, page_num)?;
45    let page_text = String::from_utf8_lossy(&page.data);
46    println!("Page Data: {}", page_text);
47
48    // Adding Data to the File
49    let data_to_add = b"This is some raw data to add to the file.";
50    page_add_data(&mut file_pointer, data_to_add)?;
51    println!("Data added to file.");
52
53    Ok(())
54}
