33pub fn init_page(page: &mut Page) {
34    // Lower offset in first 4 bytes
35    let lower = PAGE_HEADER_SIZE.to_le_bytes();
36    page.data[0..4].copy_from_slice(&lower);
37
38    // Upper offset in next 4 bytes
39    let upper = (PAGE_SIZE as u32).to_le_bytes();
40    page.data[4..8].copy_from_slice(&upper);
41}

43pub fn page_count(file: &mut File) -> io::Result<u32> {
44    // Move cursor to the start of the file header
45    file.seek(SeekFrom::Start(0))?;
46
47    // Read the first 4 bytes of the table header
48    let mut buffer = [0u8; 4];
49    file.read_exact(&mut buffer)?;
50
51    // Convert bytes to little-endian u32
52    let total_pages = u32::from_le_bytes(buffer);
53
54    Ok(total_pages)
55}

59pub fn page_free_space(page: &Page) -> io::Result<u32> {
60    // Get Lower and Upper of page
61    let lower = u32::from_le_bytes(page.data[0..4].try_into().unwrap());
62    let upper = u32::from_le_bytes(page.data[4..8].try_into().unwrap());
63    // println!("UPPER: {}", upper);
64    // println!("LOWER: {}", lower);
65    // Freespace is equal to upper - lower
66    Ok(upper - lower)
67}

69pub fn insert_tuple(file: &mut File, data: &[u8]) -> io::Result<()> {
70    // Get total number of pages in the file
71    let mut total_pages: u32 = page_count(file)?;
72    let mut last_page_num: u32 = total_pages - 1;
73    // println!("Inserting into page {}", last_page_num);
74
75    // Read last page into memory
76    let mut last_page: Page = Page::new();
77    read_page(file, &mut last_page, last_page_num)?;
78
79    // Calculate free space in the page
80    let free_space: u32 = page_free_space(&last_page)?;
81    // println!("Free space in last page: {} bytes", free_space);
82
83    // Total bytes required = tuple data + item header (offset + length)
84    let total_required = data.len() as u32 + ITEM_ID_SIZE;
85
86    if total_required > free_space {
87        // Not enough space — create a new page
88        // println!("Not enough free space in last page. Creating a new page...");
89        create_page(file)?;
90        total_pages += 1;
91        last_page_num = total_pages - 1;
92
93        // Read the newly created page (it should be empty)
94        read_page(file, &mut last_page, last_page_num)?;
95        // println!("Inserting into newly created page {}", last_page_num);
96    }
97
98    // === Insert into last_page (either old or new) ===
99
100    // Get current header offsets
101    let mut lower = u32::from_le_bytes(last_page.data[0..4].try_into().unwrap());
102    let mut upper = u32::from_le_bytes(last_page.data[4..8].try_into().unwrap());
103
104    // Compute where to place data
105    let start = upper - data.len() as u32;
106
107    // Copy tuple data into the page
108    last_page.data[start as usize..upper as usize].copy_from_slice(data);
109
110    // Update upper pointer
111    upper = start;
112    last_page.data[4..8].copy_from_slice(&upper.to_le_bytes());
113
114    // Write ItemId entry (offset + length)
115    let item_id_pos = lower as usize;
116    last_page.data[item_id_pos..item_id_pos + 4].copy_from_slice(&start.to_le_bytes());
117    last_page.data[item_id_pos + 4..item_id_pos + 8]
118        .copy_from_slice(&(data.len() as u32).to_le_bytes());
119
120    // Update lower pointer
121    lower += ITEM_ID_SIZE;
122    last_page.data[0..4].copy_from_slice(&lower.to_le_bytes());
123
124    // Write page back to disk
125    write_page(file, &mut last_page, last_page_num)?;
126    // println!("Tuple inserted successfully ({} bytes).", data.len());
127
128    Ok(())
129}

133pub fn load_csv_and_insert(
134    catalog: &Catalog,
135    db_name: &str,
136    table_name: &str,
137    file: &mut File,
138    csv_path: &str,
139) -> io::Result<()> {
140    // --- 1. Fetch table schema from catalog ---
141    let db = catalog
142        .databases
143        .get(db_name)
144        .ok_or_else(|| io::Error::new(io::ErrorKind::NotFound, format!("Database '{}' not found", db_name)))?;
145
146    let table = db
147        .tables
148        .get(table_name)
149        .ok_or_else(|| io::Error::new(io::ErrorKind::NotFound, format!("Table '{}' not found", table_name)))?;
150
151    let columns = &table.columns;
152    if columns.is_empty() {
153        return Err(io::Error::new(io::ErrorKind::InvalidData, "Table has no columns"));
154    }
155
156    // println!(
157    //     "Loading CSV '{}' into table '{}.{}' ({} columns)",
158    //     csv_path,
159    //     db_name,
160    //     table_name,
161    //     columns.len()
162    // );
163
164    // --- 2. Open and read the CSV file ---
165    let csv_file = File::open(csv_path)?;
166    let reader = BufReader::new(csv_file);
167
168    let mut lines = reader.lines();
169
170    // Skip header line
171    if let Some(Ok(header)) = lines.next() {
172        // println!("Header: {}", header);
173    }
174
175    // --- 3. Iterate through rows ---
176    let mut inserted = 0;
177    for (i, line) in lines.enumerate() {
178        let row = line?;
179        if row.trim().is_empty() {
180            continue;
181        }
182
183        // Split CSV fields by comma
184        let values: Vec<&str> = row.split(',').map(|v| v.trim()).collect();
185
186        // Validate number of columns
187        if values.len() != columns.len() {
188            println!(
189                "Skipping row {}: expected {} columns, found {}",
190                i + 1,
191                columns.len(),
192                values.len()
193            );
194            continue;
195        }
196
197        // --- 4. Serialize row based on schema ---
198        let mut tuple_bytes: Vec<u8> = Vec::new();
199
200        for (val, col) in values.iter().zip(columns.iter()) {
201            match col.data_type.as_str() {
202                "INT" => {
203                    let num: i32 = val.parse().unwrap_or_default();
204                    tuple_bytes.extend_from_slice(&num.to_le_bytes());
205                }
206                "TEXT" => {
207                    let mut text_bytes = val.as_bytes().to_vec();
208                    if text_bytes.len() > 10 {
209                        text_bytes.truncate(10);
210                    } else if text_bytes.len() < 10 {
211                        text_bytes.extend(vec![b' '; 10 - text_bytes.len()]);
212                    }
213                    tuple_bytes.extend_from_slice(&text_bytes);
214                }
215                _ => {
216                    println!(
217                        "Unsupported column type '{}' in column '{}'",
218                        col.data_type, col.name
219                    );
220                    continue;
221                }
222            }
223        }
224
225        // --- 5. Insert tuple into page system ---
226        if let Err(e) = insert_tuple(file, &tuple_bytes) {
227            println!("Failed to insert row {}: {}", i + 1, e);
228        } else {
229            inserted += 1;
230        }
231    }
232
233    
234     // --- 6️⃣. Inspect first page (for verification) ---
235    // let mut page = Page::new();
236
237    // Move to the first data page (page number 1)
238    // read_page(file, &mut page, 1)?;
239
240    // println!("\n--- First Data Page [1] Snapshot ---");
241    // println!("First 32 bytes: {:?}", &page.data[0..32]);
242    // println!("------------------------------------\n");
243    Ok(())
244}

248pub fn show_tuples(
249    catalog: &Catalog,
250    db_name: &str,
251    table_name: &str,
252    file: &mut File,
253) -> io::Result<()> {
254    // 1. Get schema from catalog
255    let db = catalog
256        .databases
257        .get(db_name)
258        .ok_or_else(|| io::Error::new(io::ErrorKind::NotFound, format!("Database '{}' not found", db_name)))?;
259
260    let table = db
261        .tables
262        .get(table_name)
263        .ok_or_else(|| io::Error::new(io::ErrorKind::NotFound, format!("Table '{}' not found", table_name)))?;
264
265    let columns = &table.columns;
266
267    // 2. Read total number of pages
268    file.seek(SeekFrom::Start(0))?;
269    let mut buf = [0u8; 4];
270    file.read_exact(&mut buf)?;
271    let mut total_pages = u32::from_le_bytes(buf);
272
273    println!("\n=== Tuples in '{}.{}' ===", db_name, table_name);
274    println!("Total pages: {}", total_pages);
275    total_pages = total_pages;
276    // 3. Loop through each page
277    for page_num in 1..total_pages {
278        let mut page = Page::new();
279        read_page(file, &mut page, page_num)?;
280        println!("\n-- Page {} --", page_num);
281
282        let lower = u32::from_le_bytes(page.data[0..4].try_into().unwrap());
283        let upper = u32::from_le_bytes(page.data[4..8].try_into().unwrap());
284        println!("Lower: {}, Upper: {}", lower, upper);
285        let num_items = (lower - PAGE_HEADER_SIZE) / ITEM_ID_SIZE;
286
287        println!("Lower: {}, Upper: {}, Tuples: {}", lower, upper, num_items);
288
289        // 4. For each tuple
290        for i in 0..num_items {
291            let base = (PAGE_HEADER_SIZE + i * ITEM_ID_SIZE) as usize;
292            let offset = u32::from_le_bytes(page.data[base..base + 4].try_into().unwrap());
293            let length = u32::from_le_bytes(page.data[base + 4..base + 8].try_into().unwrap());
294            let tuple_data = &page.data[offset as usize..(offset + length) as usize];
295
296            print!("Tuple {}: ", i + 1);
297
298            // 5. Decode each column
299            let mut cursor = 0usize;
300            for col in columns {
301                match col.data_type.as_str() {
302                    "INT" => {
303                        if cursor + 4 <= tuple_data.len() {
304                            let val = i32::from_le_bytes(tuple_data[cursor..cursor + 4].try_into().unwrap());
305                            print!("{}={} ", col.name, val);
306                            cursor += 4;
307                        }
308                    }
309                    "TEXT" => {
310                        if cursor + 10 <= tuple_data.len() {
311                            let text_bytes = &tuple_data[cursor..cursor + 10];
312                            let text = String::from_utf8_lossy(text_bytes).trim().to_string();
313                            print!("{}='{}' ", col.name, text);
314                            cursor += 10;
315                        }
316                    }
317                    _ => {
318                        print!("{}=<unsupported> ", col.name);
319                    }
320                }
321            }
322            println!();
323        }
324        // break;
325    }
326
327    println!("\n=== End of tuples ===\n");
328    Ok(())
