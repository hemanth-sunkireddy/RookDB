28pub fn init_table(file: &mut File) -> io::Result<()> {
29    // Move cursor to the beginning of the file
30    file.seek(SeekFrom::Start(0))?;
31
32    // Allocate 8192 (TABLE_HEADER_SIZE) + 8192 (PAGE_SIZE) bytes = 16KB
33    let mut zero_buf = vec![0u8; TABLE_HEADER_SIZE as usize];
34
35    //  Write "1" into the first 4 bytes (little-endian u32)
36    // This can represent the total number of pages, e.g. 1
37    zero_buf[0..4].copy_from_slice(&1u32.to_le_bytes());
38
39    // Write the full buffer (header) to the file
40    file.write_all(&zero_buf)?;
41
42    // Optionally, flush to ensure write is committed
43    file.flush()?;
44    file.sync_all()?;
45
46    create_page(file)?;
47
48    Ok(())
49}
